Version 1

uint64_t count(uint64_t n)
{
  uint64_t primes = 0;
  vector<uint8_t> is_prime(n, true);           // Initialize

  for (uint64_t k = 2; k * k < n; ++k)         // Sieve
    if (is_prime[k])
      for (uint64_t m = k * k; m < n; m += k)
        is_prime[m] = false;

  for (uint64_t k = 2; k < n; ++k)             // Count
    if (is_prime[k])
      ++primes;

  return primes;
}
